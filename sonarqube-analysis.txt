node{
   stage('SCM Checkout'){
     git 'https://github.com/shreya-17/Pixlmarketo.git'
   }
   stage('Compile-Package'){
      // Get maven home path
       sh "mvn -version"   
      sh "mvn clean install"
   }
   
   stage('SonarQube Analysis') {
        def mvnHome =  tool name: 'maven-3', type: 'maven'
        withSonarQubeEnv('sonar_server') { 
         sh "mvn clean install"
        }
    }
   }
